\chapter*{Lemas utilizados}
\addcontentsline{toc}{chapter}{Lemas utilizados}

\begin{itemize}
\item \mint{lean}|add_assoc : ∀ (a b c : ℝ), (a + b) + c = a + (b + c)|
\item \mint{lean}|add_mul : ∀ (a b c : ℝ), (a + b) * c = a * c + b * c|
\item \mint{lean}|add_zero : ∀ (a : ℝ), a + 0 = a|
\item \mint{lean}|mul_add : ∀ (a b c : ℝ), a * (b + c) = a * b + a * c|
\item \mint{lean}|mul_assoc : ∀ (a b c : ℝ), (a * b) * c = a * (b * c)|
\item \mint{lean}|mul_comm : ∀ (a b : ℝ), a * b = b * a|
\item \mint{lean}|mul_neg : ∀ (a b : ℝ), a * -b = -(a * b)|
\item \mint{lean}|mul_sub : ∀ (a b c : ℝ), a * (b - c) = a * b - a * c|
\item \mint{lean}|neg_add_self : ∀ (a : ℝ), -a + a = 0|
\item \mint{lean}|pow_two : ∀ (a : ℝ), a ^ 2 = a * a|
\item \mint{lean}|sub_eq_add_neg : ∀ (a b : ℝ), a - b = a + -b|
\item \mint{lean}|sub_self : ∀ (a : ℝ), a - a = 0|
\item \mint{lean}|two_mul : ∀ (a : ℝ), 2 * a = a + a|
\end{itemize}
